// ... (ส่วน UI เหมือนเดิม แต่ปรับ Logic การอ่าน Excel ด้านล่าง) let summaryFromExcel = { crew: 0, catering: 0, passenger: 0, total: 0 }; document.getElementById('excelInput').onchange = async (e) => { const data = await e.target.files[0].arrayBuffer(); const workbook = XLSX.read(data); const sheet = workbook.Sheets['POB']; const rows = XLSX.utils.sheet_to_json(sheet, { header: 1 }); pobData = { crew: [], catering: [], passenger: [] }; let currentSection = ""; rows.forEach(row => { // คอลัมน์: A=No, B=Cabin, C=Bed, D=Name, E=Position, F=Company const colA = String(row[0] || "").trim(); const colB = String(row[1] || "").trim(); const colC = String(row[2] || "").trim(); const colD = String(row[3] || "").trim(); // 1. ตรวจสอบการเปลี่ยน Section const textD = colD.toUpperCase(); if(textD.includes("BARGE CREW")) { currentSection = "crew"; } else if(textD.includes("CATERING")) { currentSection = "catering"; } else if(textD.includes("PASSENGER") || textD.includes("OPERATION")) { currentSection = "passenger"; } // 2. ตรวจสอบ Summary ด้านล่าง (สมมติว่ายอดรวมอยู่ในแถวที่มีคำว่า Total) if(textD === "TOTAL POB") summaryFromExcel.total = parseInt(row[4]) || 0; if(textD === "BARGE CREW" && row[4]) summaryFromExcel.crew = parseInt(row[4]) || 0; // 3. กฎเหล็ก: ต้องมี No. เป็นตัวเลข และมี Cabin, Bed, Name ครบเท่านั้นถึงจะนับเป็น POB const isNumber = !isNaN(parseInt(colA)) && /^\d+$/.test(colA); if(isNumber && colB && colC && colD && colD !== "NAME") { pobData[currentSection].push({ id: (colB + colC).replace(/\s/g, "").toUpperCase(), // เช่น 422A room: colB, name: colD.toUpperCase(), status: false, confidence: 0 }); } }); // แสดงยอดที่นับได้จริงเทียบกับ Summary const actualTotal = pobData.crew.length + pobData.catering.length + pobData.passenger.length; alert(`โหลดข้อมูลสำเร็จ! \nนับได้จริง: ${actualTotal} คน \n(Crew: ${pobData.crew.length}, Cat: ${pobData.catering.length}, Pax: ${pobData.passenger.length})`); }; // ...
